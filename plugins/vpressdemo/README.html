INTRODUCTION ​

This is a lightweight demo to demonstrate demo import and preview based on VitePress.

Usage ​

npm i -D vpressdemo shiki
1
js
// .vitepress/config.ts / config.js
import vpressdemo from "../node_modules/vpressdemo/dist";
// ...
export default {
  // ...
  markdown: {
    // ...
    config: (md) => {
      md.use(vpressdemo);
    },
  },
};
